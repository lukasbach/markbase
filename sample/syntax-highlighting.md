# Examples of Syntax Highlighting

```typescript
const func = (a: number, b: number): number => {
  return a + b;
};

const result = func(1, 2);
```

```python
def func(a: int, b: int) -> int:
  return a + b
  
result = func(1, 2)
```

```html
<div class="container">
  <h1>Heading</h1>
  <p>Paragraph</p>
</div>
```

```yaml
name: John
age: 20
```

```json
{
  "name": "John",
  "age": 20
}
```

```bash
$ echo "Hello, World!"
```

```diff
- const a = 1;
+ const a = 2;
```
